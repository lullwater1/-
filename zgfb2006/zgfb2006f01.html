*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM get_data .

  RANGES: lr_gjahr FOR zgft2006-gjahr,
          lr_zmtlg FOR zgft2006-zmtlg.

  DATA: lv_num TYPE numc3,
        lv_seq TYPE i.

  CASE gv_mode.
    WHEN gc_mode-dis.
      MOVE-CORRESPONDING: s_gjahr1[] TO lr_gjahr[],
                          s_mgrp1[] TO lr_zmtlg[].
    WHEN gc_mode-upd.
      MOVE-CORRESPONDING: s_gjahr2[] TO lr_gjahr[],
                          s_mgrp2[] TO lr_zmtlg[].
  ENDCASE.

  SELECT *
    FROM zgft2006
    INTO CORRESPONDING FIELDS OF TABLE gt_outtab
    WHERE bukrs EQ gv_bukrs
      AND gjahr IN lr_gjahr
      AND zmtlg IN lr_zmtlg.

  SORT gt_outtab BY bukrs gjahr DESCENDING zmtlg ASCENDING.

  LOOP AT gt_outtab ASSIGNING FIELD-SYMBOL(<fs_outtab>).
* --- Enter seq to use when checking duplicated key.
    lv_seq = lv_seq + 1.
    <fs_outtab>-seq = lv_seq.

    lv_num = <fs_outtab>-zconr * 100.
    SHIFT lv_num LEFT DELETING LEADING '0'.
    IF lv_num = ''.
      <fs_outtab>-zconr_c = '0'.
    ELSE.
      <fs_outtab>-zconr_c = lv_num.
    ENDIF.

    <fs_outtab>-from_db = abap_true.
  ENDLOOP.
  gv_max_seq = lv_seq.

  MOVE-CORRESPONDING gt_outtab[] TO gt_zgft2006_initial[].

  gv_saved = abap_true.

ENDFORM.

*& Form SAVE_DATA
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM save_data CHANGING pv_msg pv_typ.

  DATA: lt_zgft2006_u TYPE TABLE OF zgft2006, " updated
        lt_zgft2006_a TYPE TABLE OF zgft2006, " appended
        lt_zgft2006_d TYPE TABLE OF zgft2006, " deleted
        ls_zgrt2006   TYPE zgft2006,
        ls_outtab     TYPE gty_outtab.

* --- Check input integrity
  LOOP AT gt_outtab ASSIGNING FIELD-SYMBOL(<fs_outtab>)
                   WHERE appended = abap_true OR changed = abap_true.
    PERFORM check_if_valid USING <fs_outtab>-zconr_c
                                 <fs_outtab>-gjahr
                        CHANGING <fs_outtab>-zconr
                                 <fs_outtab>-icon
                                 <fs_outtab>-msg
                                 <fs_outtab>-savable.
  ENDLOOP.

* --- Get taqrget lists by purposes
  PERFORM get_changed_list TABLES lt_zgft2006_u
                                  lt_zgft2006_a
                                  lt_zgft2006_d.

* --- SAVE: UPDATE
  ls_zgrt2006-bukrs = gv_bukrs.
  ls_zgrt2006-erdat = sy-datum.
  ls_zgrt2006-ernam = sy-uname.
  ls_zgrt2006-erzet = sy-uzeit.
  MODIFY lt_zgft2006_u FROM ls_zgrt2006
                            TRANSPORTING bukrs erdat ernam erzet
                            WHERE zmtlg <> ''.
  MODIFY zgft2006 FROM TABLE lt_zgft2006_u.

* --- SAVE: INSERT(APPENDED)
  ls_zgrt2006-bukrs = gv_bukrs.
  ls_zgrt2006-aedat = sy-datum.
  ls_zgrt2006-aenam = sy-uname.
  ls_zgrt2006-aezet = sy-uzeit.
  MODIFY lt_zgft2006_a FROM ls_zgrt2006
                            TRANSPORTING bukrs aedat aenam aezet
                            WHERE zmtlg <> ''.
  MODIFY zgft2006 FROM TABLE lt_zgft2006_a.

* --- DELETE
  IF lt_zgft2006_d[] IS NOT INITIAL.
    DELETE zgft2006 FROM TABLE lt_zgft2006_d.
  ENDIF.

  IF lt_zgft2006_u IS INITIAL AND
    lt_zgft2006_a IS INITIAL AND
    lt_zgft2006_d IS INITIAL.
    pv_msg = TEXT-t25.
    pv_typ = 'E'.
  ELSE.
    pv_msg = TEXT-t24.
    pv_typ = 'S'.
  ENDIF.

ENDFORM.
