FORM MAKE_INTERNAL .

  DATA: LT_DREF TYPE REF TO DATA,
        LS_DREF TYPE REF TO DATA.

  CALL FUNCTION 'DDIF_NAMETAB_GET'
    EXPORTING
      TABNAME   = P_TAB
    IMPORTING
      DFIES_WA  = GS_DFIES_WA

      DFIES_TAB = GT_DFIES
    EXCEPTIONS
      NOT_FOUND = 1
      OTHERS    = 2.

  IF SY-SUBRC <> 0.
    MESSAGE ID SY-MSGID TYPE SY-MSGTY NUMBER SY-MSGNO

  ENDIF.

  LOOP AT GT_DFIES.

    MOVE: GT_DFIES-FIELDNAME  TO GS_CAT-FIELDNAME,
          GT_DFIES-POSITION   TO GS_CAT-COL_POS,
          'g'                 TO GS_CAT-INTTYPE,
          'STRG'              TO GS_CAT-DATATYPE,
          GT_DFIES-OUTPUTLEN  TO GS_CAT-INTLEN.
    APPEND GS_CAT TO GT_LVC_CAT .
    CLEAR: GS_CAT.

  ENDLOOP.

  CALL METHOD CL_ALV_TABLE_CREATE=>CREATE_DYNAMIC_TABLE
    EXPORTING

      OTHERS                    = 2.

  IF SY-SUBRC <> 0.
    MESSAGE ID SY-MSGID TYPE SY-MSGTY NUMBER SY-MSGNO
    WITH SY-MSGV1 SY-MSGV2 SY-MSGV3 SY-MSGV4.
  ENDIF.

  ASSIGN GP_TABLE->* TO <GT_OUTTAB>.

  CREATE DATA GP_STRUCT LIKE LINE OF <GT_OUTTAB>.
  ASSIGN GP_STRUCT->* TO <GS_OUTTAB>.

  CREATE DATA LT_DREF TYPE TABLE OF (P_TAB).
  CREATE DATA LS_DREF TYPE (P_TAB).
