  CASE ok_code.
    WHEN 'REFRE'.
      PERFORM refresh_data.

    WHEN 'AREFRE'.
      PERFORM alv_table_refresh USING 'GRID'.

    WHEN 'SAVE'.
      PERFORM save_data.

    WHEN 'BACK' OR 'CANCEL'.
      SET SCREEN 0.

    WHEN 'EXIT'.
      IF grid IS NOT INITIAL AND
         cont IS NOT INITIAL.
        CALL METHOD : grid->free,
                      cont->free.
        FREE : grid, cont.
      ENDIF.
      LEAVE PROGRAM.
  ENDCASE.
