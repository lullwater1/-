FORM GET_DATA .

  CLEAR: GT_OUTTAB, GT_OUTTAB[], GT_OUTTAB_INITIAL, GT_OUTTAB_INITIAL[].
  SELECT *
    FROM ZGFT2007
    INTO CORRESPONDING FIELDS OF TABLE GT_OUTTAB
    ORDER BY PRIMARY KEY.

  CHECK LINES( GT_OUTTAB[] ) > 0.

  MOVE-CORRESPONDING GT_OUTTAB[] TO GT_OUTTAB_INITIAL[].

  GS_OUTTAB-FROM_DB = ABAP_TRUE.
  MODIFY GT_OUTTAB[] FROM GS_OUTTAB TRANSPORTING FROM_DB WHERE MATNR <> '' .

  GV_EDITED = ABAP_FALSE.   " TO INITIALLY DISABLE SAVE BUTTON

ENDFORM.

FORM SAVE_DATA CHANGING PV_MODIFIED PV_DELETED.

  DATA: LT_OUTTAB_COPY    TYPE TABLE OF GTY_OUTTAB WITH HEADER LINE,
        LT_OUTTAB         TYPE TABLE OF GTY_OUTTAB WITH HEADER LINE,
        LT_OUTTAB_DELETED TYPE TABLE OF GTY_OUTTAB,
        LT_2B_DELETED     TYPE TABLE OF ZGFT2007,
        LT_2B_SAVED       TYPE TABLE OF ZGFT2007 WITH HEADER LINE.

* ---------------------
* --- Get deleted
* ---------------------
  MOVE-CORRESPONDING GT_OUTTAB[] TO LT_OUTTAB_COPY[].
  SORT LT_OUTTAB_COPY BY MATNR PFONA DATAB.

  DELETE LT_OUTTAB_COPY WHERE APPENDED = ABAP_TRUE.
  CLEAR: GT_OUTTAB_INITIAL.
  LOOP AT GT_OUTTAB_INITIAL.
    READ TABLE LT_OUTTAB_COPY WITH KEY MATNR = GT_OUTTAB_INITIAL-MATNR
                                       PFONA = GT_OUTTAB_INITIAL-PFONA
                                       DATAB = GT_OUTTAB_INITIAL-DATAB
                                       BINARY SEARCH.
    CHECK SY-SUBRC IS NOT INITIAL.

    APPEND GT_OUTTAB_INITIAL TO LT_OUTTAB_DELETED.
    CLEAR: GT_OUTTAB_INITIAL.
  ENDLOOP.

  MOVE-CORRESPONDING LT_OUTTAB_DELETED[] TO  LT_2B_DELETED[].

* ---------------------
* --- Get changed
* ---------------------
* All data in the DB table are displayed in the grid
* ( since no search criteria set initially),
* means that there's no hidden record to consider
* when checking data integrity for saving

* -------------------------
* --- Check data integrity
* -------------------------
* 1.1 to see if there's duplicated key,
* 1.2 START/END DATE ORDER
  DATA: LV_ROW_CNT TYPE I.

  CLEAR: LT_OUTTAB_COPY[], LT_OUTTAB_COPY, LT_OUTTAB, LT_OUTTAB[].
  MOVE-CORRESPONDING: GT_OUTTAB[] TO LT_OUTTAB[].
  LOOP AT LT_OUTTAB.

    IF LT_OUTTAB-BUKRS IS INITIAL OR
      LT_OUTTAB-MATNR IS INITIAL OR
      LT_OUTTAB-PFONA IS INITIAL OR
      ( LT_OUTTAB-DATAB IS INITIAL OR LT_OUTTAB-DATAB = '00000000' ) .
      LT_OUTTAB_COPY-MSG = TEXT-T11.
      LT_OUTTAB_COPY-INPUT_ERR = ABAP_TRUE.
      MODIFY GT_OUTTAB FROM LT_OUTTAB_COPY TRANSPORTING MSG INPUT_ERR
        WHERE BUKRS = LT_OUTTAB-BUKRS
          AND MATNR = LT_OUTTAB-MATNR
          AND PFONA = LT_OUTTAB-PFONA
          AND DATAB = LT_OUTTAB-DATAB
          AND FROM_DB = ABAP_FALSE
          AND ( CHANGED = ABAP_TRUE OR APPENDED = ABAP_TRUE ).
      CONTINUE.
    ENDIF.

    MOVE-CORRESPONDING GT_OUTTAB[] TO LT_OUTTAB_COPY[].
    DELETE LT_OUTTAB_COPY WHERE BUKRS <> LT_OUTTAB-BUKRS.
    DELETE LT_OUTTAB_COPY WHERE MATNR <> LT_OUTTAB-MATNR.
    DELETE LT_OUTTAB_COPY WHERE PFONA <> LT_OUTTAB-PFONA.
    DELETE LT_OUTTAB_COPY WHERE DATAB <> LT_OUTTAB-DATAB.

    LV_ROW_CNT = LINES( LT_OUTTAB_COPY[] ).

    IF LV_ROW_CNT > 1.
      LT_OUTTAB_COPY-MSG = TEXT-T08.
      LT_OUTTAB_COPY-INPUT_ERR = ABAP_TRUE.
      MODIFY GT_OUTTAB FROM LT_OUTTAB_COPY TRANSPORTING MSG INPUT_ERR
        WHERE BUKRS = LT_OUTTAB-BUKRS
          AND MATNR = LT_OUTTAB-MATNR
          AND PFONA = LT_OUTTAB-PFONA
          AND DATAB = LT_OUTTAB-DATAB
          AND FROM_DB = ABAP_FALSE.
    ELSE.
      IF LT_OUTTAB-DATAB > LT_OUTTAB-DATBI.
        LT_OUTTAB_COPY-MSG = TEXT-T10.
        LT_OUTTAB_COPY-INPUT_ERR = ABAP_TRUE.
        MODIFY GT_OUTTAB FROM LT_OUTTAB_COPY TRANSPORTING MSG INPUT_ERR
          WHERE BUKRS = LT_OUTTAB-BUKRS
            AND MATNR = LT_OUTTAB-MATNR
            AND PFONA = LT_OUTTAB-PFONA
            AND DATAB = LT_OUTTAB-DATAB
            AND FROM_DB = ABAP_FALSE.
      ENDIF.
    ENDIF.

    CLEAR: LT_OUTTAB, LT_OUTTAB_COPY[], LT_OUTTAB_COPY.
  ENDLOOP.

* 2. to see if there's overlapped period
***  SORT LT_OUTTAB_COPY BY MATNR PFONA DATAB.
***
***  DELETE LT_OUTTAB_COPY WHERE FROM_DB = ABAP_TRUE AND CHANGED = ABAP_FALSE.
***  DELETE LT_OUTTAB_COPY WHERE APPENDED = ABAP_FALSE AND CHANGED = ABAP_FALSE.
  CLEAR: LT_OUTTAB_COPY[], LT_OUTTAB_COPY, LT_OUTTAB, LT_OUTTAB[].
  MOVE-CORRESPONDING GT_OUTTAB[] TO LT_OUTTAB[].
  LOOP AT LT_OUTTAB.
    " 1. DATAB: IN BETWEEN
    " 3. BOTH ARE INCLUDED
    " REMOVE GOOD RECORDS
    MOVE-CORRESPONDING GT_OUTTAB[] TO LT_OUTTAB_COPY[].
    DELETE LT_OUTTAB_COPY WHERE BUKRS <> LT_OUTTAB-BUKRS.
    DELETE LT_OUTTAB_COPY WHERE MATNR <> LT_OUTTAB-MATNR.
    DELETE LT_OUTTAB_COPY WHERE PFONA <> LT_OUTTAB-PFONA.

    DELETE LT_OUTTAB_COPY WHERE DATBI < LT_OUTTAB-DATAB.
    DELETE LT_OUTTAB_COPY WHERE DATAB > LT_OUTTAB-DATBI.
    LV_ROW_CNT = LINES( LT_OUTTAB_COPY[] ).

    IF LV_ROW_CNT > 1.
      LT_OUTTAB_COPY-MSG = TEXT-T09.
      LT_OUTTAB_COPY-INPUT_ERR = ABAP_TRUE.
      MODIFY GT_OUTTAB FROM LT_OUTTAB_COPY TRANSPORTING MSG INPUT_ERR
        WHERE BUKRS = LT_OUTTAB-BUKRS
          AND MATNR = LT_OUTTAB-MATNR
          AND PFONA = LT_OUTTAB-PFONA
          AND DATAB = LT_OUTTAB-DATAB
          AND CHANGED = ABAP_TRUE.
    ENDIF.

    CLEAR: LT_OUTTAB, LT_OUTTAB_COPY[], LT_OUTTAB_COPY.
  ENDLOOP.

* --- all are okay
  MOVE-CORRESPONDING GT_OUTTAB[] TO LT_OUTTAB_COPY[].
  DELETE LT_OUTTAB_COPY WHERE INPUT_ERR = ABAP_TRUE.
  DELETE LT_OUTTAB_COPY WHERE FROM_DB = ABAP_TRUE AND CHANGED <> ABAP_TRUE .
  MOVE-CORRESPONDING LT_OUTTAB_COPY[] TO  LT_2B_SAVED[].

  " Time stamp
  LT_2B_SAVED-ERDAT = LT_2B_SAVED-AEDAT = SY-DATUM.
  LT_2B_SAVED-ERZET = LT_2B_SAVED-AEZET = SY-UZEIT.
  LT_2B_SAVED-ERNAM = LT_2B_SAVED-AENAM = SY-UNAME.
  MODIFY LT_2B_SAVED TRANSPORTING ERDAT
                                  ERZET
                                  ERNAM
                                  AEDAT
                                  AEZET
                                  AENAM WHERE BUKRS <> ''.

* ---------------------
* --- Save/Delete
* ---------------------
  DATA: LV_MODIFIED TYPE NUMC5,
        LV_DELETED  TYPE NUMC5.
  LV_MODIFIED = LINES( LT_2B_SAVED[] ).
  LV_DELETED = LINES( LT_2B_DELETED[] ).

  IF LV_DELETED > 0.
    SHIFT LV_DELETED LEFT DELETING LEADING '0'.
    PV_DELETED = LV_DELETED.
  ELSE.
    PV_DELETED = '0'.
  ENDIF.

  IF LV_MODIFIED > 0.
    SHIFT LV_MODIFIED LEFT DELETING LEADING '0'.
    PV_MODIFIED = LV_MODIFIED.
  ELSE.
    PV_MODIFIED = '0'.
  ENDIF.

  MODIFY ZGFT2007 FROM TABLE LT_2B_SAVED[].
  DELETE ZGFT2007 FROM TABLE LT_2B_DELETED[].

ENDFORM.

FORM INITIALIZE_VARS .

  DATA: LS_OUTTAB TYPE GTY_OUTTAB.

  LS_OUTTAB-FROM_DB = ABAP_TRUE.
  MODIFY GT_OUTTAB FROM LS_OUTTAB TRANSPORTING APPENDED
                                               INPUT_ERR
                                               DELETED
                                               CHANGED
                                               FROM_DB
                                WHERE APPENDED <> '' OR
                                      INPUT_ERR <> '' OR
                                      CHANGED <> '' OR
                                      DELETED <> ''.

  GV_EDITED = ABAP_FALSE.

ENDFORM.

FORM CHECK_IF_CHANGED CHANGING PV_ANSWER.

***  CHECK GT_OUTTAB_DEL IS INITIAL.

  READ TABLE GT_OUTTAB WITH KEY DELETED = ABAP_TRUE INTO GS_OUTTAB.

  READ TABLE GT_OUTTAB WITH KEY APPENDED = ABAP_TRUE INTO GS_OUTTAB.

  READ TABLE GT_OUTTAB WITH KEY CHANGED = ABAP_TRUE INTO GS_OUTTAB.
