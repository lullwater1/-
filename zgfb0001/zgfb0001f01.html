FORM DATA_SAVE .

  DATA: LT_ZKGT0001  TYPE TABLE OF ZGFT0001  WITH HEADER LINE,
        LT_ZKGT0001T TYPE TABLE OF ZGFT0001T WITH HEADER LINE.

  DATA: L_OREF TYPE REF TO CX_ROOT,
        L_TEXT TYPE STRING.

  CLEAR G_COUNT.

  FIELD-SYMBOLS <FS1> TYPE ANY.

  DATA LR_SELECTIONS TYPE REF TO CL_SALV_SELECTIONS.

  DATA LT_ROWS  TYPE SALV_T_ROW.
  DATA L_ROW    TYPE I.

  LR_SELECTIONS = GO_ALV->GET_SELECTIONS( ).
  LT_ROWS = LR_SELECTIONS->GET_SELECTED_ROWS( ).

  LOOP AT LT_ROWS INTO L_ROW.

    ADD 1 TO G_COUNT.

    READ TABLE GT_OUTTAB INTO GS_OUTTAB INDEX L_ROW.

    IF SY-SUBRC = 0.
      READ TABLE GT_ZKGT0001 INTO GS_ZKGT0001
                             WITH  KEY MANDT = SY-MANDT
                                            PFONA = GS_OUTTAB-PFONA
                                            CCNGN = GS_OUTTAB-CCNGN
                                            RSEQ  = GS_OUTTAB-RSEQ.

      IF SY-SUBRC = 0.
*-- ## #### ## ## ##
        MOVE GS_ZKGT0001 TO LT_ZKGT0001.
        MOVE GS_ZKGT0001-ERDAT TO GS_OUTTAB-ERDAT.
        MOVE GS_ZKGT0001-ERZET TO GS_OUTTAB-ERZET.
        MOVE GS_ZKGT0001-ERNAM TO GS_OUTTAB-ERNAM.

        MOVE-CORRESPONDING GS_OUTTAB TO LT_ZKGT0001.
        ASSIGN LT_ZKGT0001 TO <FS1>.
        PERFORM SET_CREATE_CHANGE_DATA(ZGF_FORM_SUB_ROUTINE)
                                       USING 'M'
                                       CHANGING <FS1>.

      ELSE.
*-- ## #### ## ## ##
        MOVE-CORRESPONDING GS_OUTTAB TO LT_ZKGT0001.
        ASSIGN LT_ZKGT0001 TO <FS1>.
        PERFORM SET_CREATE_CHANGE_DATA(ZGF_FORM_SUB_ROUTINE)
                                       USING 'A'
                                       CHANGING <FS1>.

      ENDIF.

      APPEND LT_ZKGT0001.
      CLEAR  LT_ZKGT0001.

*-- ###
      MOVE-CORRESPONDING GS_OUTTAB TO LT_ZKGT0001T.
      MOVE SY-LANGU                TO LT_ZKGT0001T-SPRAS.
      APPEND LT_ZKGT0001T.
      CLEAR  LT_ZKGT0001T.

    ENDIF.

  ENDLOOP.

  IF SY-SUBRC <> 0.
    MESSAGE I005 DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.

  TRY.
      CLEAR L_TEXT.

      MODIFY ZGFT0001  FROM TABLE LT_ZKGT0001.
      MODIFY ZGFT0001T FROM TABLE LT_ZKGT0001T.

      COMMIT WORK.
      MESSAGE I003 WITH G_COUNT.
      MOVE ICON_LED_GREEN TO GS_OUTTAB-ICON.
      MODIFY GT_OUTTAB FROM GS_OUTTAB TRANSPORTING ICON
                                      WHERE ICON <> ICON_LED_GREEN.

    CATCH CX_SY_OPEN_SQL_DB INTO L_OREF.
      L_TEXT = L_OREF->GET_TEXT( ).

      ROLLBACK WORK.
      MESSAGE I004 WITH L_TEXT DISPLAY LIKE 'E'.
      MOVE ICON_LED_RED TO GS_OUTTAB-ICON.
      MODIFY GT_OUTTAB FROM GS_OUTTAB TRANSPORTING ICON
                                      WHERE ICON <> ICON_LED_RED.

  ENDTRY.

  CALL METHOD GO_ALV->REFRESH
    EXPORTING
      REFRESH_MODE = IF_SALV_C_REFRESH=>SOFT.

ENDFORM. " DATA_SAVE

FORM F4_PFONA USING IV_DYNPROFIELD TYPE HELP_INFO-DYNPROFLD.

  TYPES: BEGIN OF TY_F4_PFONA,
           PFONA TYPE ZGFT0001-PFONA,
           FTANA TYPE ZGFT0004-FTANA,
         END OF TY_F4_PFONA.

  DATA: LV_BUKRS  TYPE BUKRS,
        LT_VALUES TYPE TABLE OF DYNPREAD,
        LS_VALUE  TYPE DYNPREAD,
        LT_RET    TYPE TABLE OF DDSHRETVAL,
        LT_PFONA  TYPE STANDARD TABLE OF TY_F4_PFONA WITH EMPTY KEY.

  CLEAR: LT_VALUES, LS_VALUE, LV_BUKRS.
  LS_VALUE-FIELDNAME = 'P_BUKRS'.
  APPEND LS_VALUE TO LT_VALUES.

  CALL FUNCTION 'DYNP_VALUES_READ'
    EXPORTING
      DYNAME               = SY-REPID

      INVALID_DYNPRONAME   = 3

  READ TABLE LT_VALUES INTO LS_VALUE WITH KEY FIELDNAME = 'P_BUKRS'.
