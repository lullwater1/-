FUNCTION ZGF_PRT_DO_PL.
*"----------------------------------------------------------------------
*"*"Local Interface:
*" IMPORTING
*" VALUE(I_BUKRS) TYPE BUKRS OPTIONAL
*" VALUE(I_PARTNER) TYPE LIFNR OPTIONAL
*" VALUE(I_CO_NUMBER) TYPE ZKGE_CCNO
*" VALUE(I_OBJID) TYPE W3OBJID OPTIONAL
*" EXPORTING
*" VALUE(E_RETURN) TYPE CHAR1
*" VALUE(E_RETURNMSG) TYPE CHAR100
*" TABLES
*" ET_ZGFS0124 STRUCTURE ZGFS0124 OPTIONAL
*"----------------------------------------------------------------------
* Global data declarations

  DATA: LS_ITEM LIKE ZGFS0124,
        LT_ITEM LIKE TABLE OF LS_ITEM    WITH HEADER LINE.

  DATA: LV_KUNAG TYPE ZGFT0550-KUNAG,
        LV_DETP  TYPE ZKGE_DETP,
        LV_PRETP TYPE ZKGE_PRETP.

  DATA: LS_ZGFT0014 TYPE ZGFT0014.

*********************** Supplier *******************************
  SELECT SINGLE KUNAG DETP PRETP
    INTO (LV_KUNAG, LV_DETP, LV_PRETP)
    FROM ZGFT0550
   WHERE CCNO  EQ I_CO_NUMBER.

*********************** Item List ******************************
*** SELECT A~KUNAG
*** A~PFONA
*** A~CCNGN
*** A~MATNR
*** A~KDMAT
*** A~MAKTX
*** A~CCNGN
*** A~COFLGBF
*** A~DATAB
*** A~DATBI
*** A~CCNO
*** C~LAND_TEXT
***
*** INTO CORRESPONDING FIELDS OF TABLE LT_ITEM
*** FROM ZGFT0550 AS A INNER JOIN ZGFT0014 AS C
*** ON A~BUKRS = C~BUKRS
*** WHERE A~CCNO EQ I_CO_NUMBER
*** AND A~PRETP EQ LV_PRETP
*** AND A~DETP EQ LV_DETP.

  SELECT A~KUNAG
         A~PFONA
         A~MATNR
         A~KDMAT
         A~MAKTX
         A~CCNGN
         A~COFLGBF
         A~DATAB
         A~DATBI
         A~CCNO
   INTO CORRESPONDING FIELDS OF TABLE LT_ITEM
   FROM ZGFT0550 AS A
  WHERE A~CCNO    EQ I_CO_NUMBER
    AND A~PRETP   EQ LV_PRETP
    AND A~DETP    EQ LV_DETP.

  CHECK SY-SUBRC IS INITIAL.

  SELECT SINGLE *
    FROM ZGFT0014
    INTO CORRESPONDING FIELDS OF LS_ZGFT0014
    WHERE BUKRS = I_BUKRS
      AND LAND_TEXT NE ''.

  IF SY-SUBRC IS INITIAL.
    LS_ITEM-LAND_TEXT = LS_ZGFT0014-LAND_TEXT.
    MODIFY LT_ITEM FROM LS_ITEM TRANSPORTING LAND_TEXT
                                    WHERE CCNO = I_CO_NUMBER.
  ENDIF.

  ET_ZGFS0124[] = LT_ITEM[].

ENDFUNCTION.
