Description: [Global FTA] Exract Window List Box Value
FUNCTION ZGF_LISTBOX_DATA.
*"----------------------------------------------------------------------
*"*"Local Interface:
*" IMPORTING
*" VALUE(I_SGUBUN) TYPE ZKGE_SGUBUN
*" VALUE(I_SPRAS) TYPE SPRAS DEFAULT SY-LANGU
*" TABLES
*" T_LISTVALUE STRUCTURE ZGFS0050
*" EXCEPTIONS
*" LIST_VALUE_NOT_FOUND
*"----------------------------------------------------------------------

* Global data declarations

  CLEAR: T_LISTVALUE, T_LISTVALUE[].

  SELECT A~SGUBUN
         A~SVALUE
         B~SVTXT
         A~SPOS
    INTO CORRESPONDING FIELDS OF TABLE T_LISTVALUE
    FROM ZGFT5000M AS A LEFT OUTER JOIN ZGFT5000MT AS B
                   ON A~SGUBUN EQ B~SGUBUN
                  AND A~SVALUE EQ B~SVALUE
                  AND B~SPRAS  EQ I_SPRAS
   WHERE A~SGUBUN EQ I_SGUBUN.

  IF SY-SUBRC <> 0.
    RAISE LIST_VALUE_NOT_FOUND.
    EXIT.
  ENDIF.

  SORT T_LISTVALUE BY SPOS.

ENDFUNCTION.
