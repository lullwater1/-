FORM TAB_SCREEN_SET .

  TAB1 = TEXT-N01.
  TAB2 = TEXT-N02.

  IF SY-BATCH = ZGFTP_TRUE.
    TABB1-DYNNR     = 102.
    TABB1-ACTIVETAB = 'PUSH2'.
    TABB1-PROG = SY-REPID.
    G_DYNNR = TABB1-DYNNR.
    RETURN.
  ENDIF.

  PERFORM GET_LATEST_TAB(ZGF_FORM_SUB_ROUTINE) USING SY-REPID
                                            CHANGING G_DYNNR
                                                   TABB1-ACTIVETAB.

  IF G_DYNNR IS INITIAL.
    TABB1-DYNNR     = 101.
    TABB1-ACTIVETAB = 'PUSH1'.
    G_DYNNR = TABB1-DYNNR.
  ELSE.
    TABB1-DYNNR     = G_DYNNR.
  ENDIF.

  TABB1-PROG      = SY-REPID.


ENDFORM. " TAB_SCREEN_SET

FORM TAB_SCREEN_POSTION_SET .

  G_UCOMM = SY-UCOMM.

      TABB1-ACTIVETAB = 'PUSH2'.
      G_DYNNR = TABB1-DYNNR.
    WHEN OTHERS.
  ENDCASE.

  PERFORM SET_LATEST_TAB(ZGF_FORM_SUB_ROUTINE) USING SY-REPID
                                                     G_DYNNR
                                                     TABB1-ACTIVETAB.

FORM GET_TABLE_DATA.

  CLEAR: GT_OUTTAB, GT_OUTTAB[].

  SELECT *
    INTO CORRESPONDING FIELDS OF TABLE GT_OUTTAB
    FROM ZGFT0480
   WHERE BUKRS IN S_BUKR2
     AND MTROH IN S_MATN2
     AND DATUV IN S_DATU2.

  IF GT_OUTTAB[] IS INITIAL.
    MESSAGE S001 DISPLAY LIKE 'E'.
    STOP.
  ENDIF.

ENDFORM. "GET_TABLE_DATA

FORM HANDLE_DOUBLE_CLICK USING P_ROW
                                P_COLUMN.
  DATA: GS_OUTTAB TYPE ZGFT0480.

  READ TABLE GT_OUTTAB INTO GS_OUTTAB INDEX P_ROW.
  IF SY-SUBRC = 0.
    CASE P_COLUMN.
      WHEN 'MTROH'.
        CHECK GS_OUTTAB-MTROH IS NOT INITIAL.
        SET PARAMETER ID 'MAT' FIELD GS_OUTTAB-MTROH.
        SET PARAMETER ID 'MXX' FIELD 'K'.
        CALL TRANSACTION 'MM03' AND SKIP FIRST SCREEN.

    ENDCASE.
  ENDIF.

ENDFORM. " HANDLE_USER_COMMAND
