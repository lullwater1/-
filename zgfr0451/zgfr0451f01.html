FORM SET_SCREEN .

  LOOP AT SCREEN.
    IF SCREEN-NAME = 'P_PALL'.
      SCREEN-ACTIVE = '0'.
      MODIFY SCREEN.
    ENDIF.
  ENDLOOP.

  CASE P_PALL.

    WHEN ZGFTP_TRUE.

      LOOP AT SCREEN.

        IF SCREEN-GROUP1 = 'MD2'.
          SCREEN-ACTIVE = 1.

        ELSEIF SCREEN-GROUP1 = 'MD1'.
          SCREEN-ACTIVE = 0.
        ENDIF.

        MODIFY SCREEN.

      ENDLOOP.

    WHEN ZGFTP_FALSE.

      LOOP AT SCREEN.

        IF SCREEN-GROUP1 = 'MD1'.
          SCREEN-ACTIVE = 0.

        ELSEIF SCREEN-GROUP1 = 'MD2'.
          SCREEN-ACTIVE = 0.
        ENDIF.

        MODIFY SCREEN.

      ENDLOOP.

  ENDCASE.
