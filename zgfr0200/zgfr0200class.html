FORM handle_double_click USING p_row
                                p_column.

  DATA: gs_outtab TYPE zgft0200.

  READ TABLE gt_outtab INTO gs_outtab INDEX p_row.
  IF sy-subrc = 0.
    CASE p_column.
      WHEN 'MJAHR' OR 'MBLNR' OR 'ZEILE'.
        CHECK gs_outtab-mblnr IS NOT INITIAL.
        CHECK gs_outtab-mjahr IS NOT INITIAL.
        CHECK gs_outtab-zeile IS NOT INITIAL.

        CALL FUNCTION 'MIGO_DIALOG'
          EXPORTING
            i_action            = 'A04'
            i_refdoc            = 'R02'
            i_notree            = 'X'
            i_no_auth_check     = ' '
            i_deadend           = 'X'
            i_skip_first_screen = 'X'
            i_okcode            = 'OK_GO'
            i_mblnr             = gs_outtab-mblnr
            i_mjahr             = gs_outtab-mjahr
            i_zeile             = gs_outtab-zeile.

      WHEN 'EBELN' OR 'EBELP'.
        CHECK gs_outtab-ebeln IS NOT INITIAL.
        CHECK gs_outtab-ebelp IS NOT INITIAL.

        SET PARAMETER ID 'BES' FIELD gs_outtab-ebeln.
        SET PARAMETER ID 'BSP' FIELD gs_outtab-ebelp.
        CALL TRANSACTION 'ME23N' AND SKIP FIRST SCREEN.

      WHEN 'LIFNR'.
        CHECK gs_outtab-lifnr IS NOT INITIAL.

        SET PARAMETER ID 'LIF' FIELD gs_outtab-lifnr.
        SET PARAMETER ID 'KDY' FIELD '/110/120'.
        CALL TRANSACTION 'XK03' AND SKIP FIRST SCREEN.

      WHEN 'MATNR'.
        CHECK gs_outtab-matnr IS NOT INITIAL.

        SET PARAMETER ID 'MAT' FIELD gs_outtab-matnr.
        SET PARAMETER ID 'MXX' FIELD 'K'.

FORM get_table_data.

  CLEAR: gt_outtab, gt_outtab[].

*  SELECT *
*    INTO CORRESPONDING FIELDS OF TABLE GT_OUTTAB
*    FROM ZGFT0200
*   WHERE BUDAT IN S_BUDAT
*     AND BUKRS  = P_BUKR2
*     AND WERKS IN S_WERK2.
  SELECT a~*,
         b~name1 AS lifnm,
         c~maktx AS matnm,
         d~z_exct
    INTO CORRESPONDING FIELDS OF TABLE @gt_outtab
    FROM zgft0200 AS a
    LEFT JOIN lfa1 AS b
           ON a~lifnr = b~lifnr
    LEFT JOIN makt AS c
           ON a~matnr = c~matnr
          AND c~spras = @sy-langu
    LEFT JOIN zgft2010 AS d
           ON a~bukrs = d~bukrs
          AND a~werks = d~werks
          AND a~matnr = d~matnr
   WHERE a~budat IN @s_budat
     AND a~bukrs =  @p_bukr2
     AND a~werks IN @s_werk2.

  IF gt_outtab[] IS INITIAL.
    MESSAGE s001 DISPLAY LIKE 'E'.
    STOP.
  ENDIF.

ENDFORM. "GET_TABLE_DATA
