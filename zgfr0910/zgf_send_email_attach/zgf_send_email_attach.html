Description: [Global FTA] Send e-mail(file attachment)
FUNCTION ZGF_SEND_EMAIL_ATTACH.
*"----------------------------------------------------------------------
*"*"Local Interface:
*" IMPORTING
*" VALUE(I_TITLE) TYPE SO_OBJ_DES
*" VALUE(I_SENDER) TYPE ZGFS0201 OPTIONAL
*" VALUE(I_TYPE) TYPE SO_OBJ_TP DEFAULT 'HTM'
*" VALUE(I_REQUESTED_STATUS) TYPE BCS_RQST DEFAULT 'N'
*" VALUE(I_STATUS_MAIL) TYPE BCS_STML DEFAULT 'N'
*" VALUE(I_SPRAS) TYPE SPRAS DEFAULT SY-LANGU
*" VALUE(I_ATTFG) TYPE CHAR01 DEFAULT ' '
*" VALUE(IS_ATTACH) TYPE ZGFT7005 OPTIONAL
*" EXPORTING
*" VALUE(E_RETURN) TYPE BAPIRET2
*" TABLES
*" T_MAILRECIPIENTS STRUCTURE ZGFS0202
*" T_MAILCONTENTS TYPE SOLI_TAB
*"----------------------------------------------------------------------

* Global data declarations

  DATA: SEND_REQUEST    TYPE REF TO CL_BCS,
        BCS_EXCEPTION   TYPE REF TO CX_BCS.

  DATA DOCUMENT         TYPE REF TO CL_DOCUMENT_BCS.
  DATA SENDER           TYPE REF TO IF_SENDER_BCS.
  DATA RECIPIENT        TYPE REF TO IF_RECIPIENT_BCS.
  DATA SAPSENDER        TYPE REF TO CL_SAPUSER_BCS.
  DATA SENT_TO_ALL      TYPE OS_BOOLEAN.

  DATA: L_FILENAME  TYPE SOOD-OBJDES,
        L_EXTENSION TYPE SOODK-OBJTP,
        L_FILELEN   TYPE SOOD-OBJLEN.

  DATA LT_CONTENTS_BIN  TYPE SOLIX_TAB.

  DATA L_MESSAGE TYPE STRING.

  TRY.
      SEND_REQUEST = CL_BCS=>CREATE_PERSISTENT( ).

*-- ##/ ###
      DOCUMENT = CL_DOCUMENT_BCS=>CREATE_DOCUMENT(
        I_TYPE    = I_TYPE
        I_TEXT    = T_MAILCONTENTS[]
        I_SUBJECT = I_TITLE ).

*-- ####
      IF I_ATTFG = ZGFTP_TRUE.

        CL_BCS_CONVERT=>XSTRING_TO_SOLIX(
             EXPORTING
               IV_XSTRING   = IS_ATTACH-FILEOBJECT
             RECEIVING
               ET_SOLIX     = LT_CONTENTS_BIN ).

        L_EXTENSION = IS_ATTACH-FILEEXT.
        L_FILENAME  = IS_ATTACH-FILENAME.
        L_FILELEN   = IS_ATTACH-FILELENGTH.

        CALL METHOD DOCUMENT->ADD_ATTACHMENT
          EXPORTING
            I_ATTACHMENT_TYPE    = L_EXTENSION
            I_ATTACHMENT_SUBJECT = L_FILENAME
            I_ATTACHMENT_SIZE    = L_FILELEN
            I_ATT_CONTENT_HEX    = LT_CONTENTS_BIN.

      ENDIF.

      SEND_REQUEST->SET_DOCUMENT( DOCUMENT ).

*-- ### ###
      IF I_SENDER-SEND_ADDR IS NOT INITIAL.

        CALL METHOD CL_CAM_ADDRESS_BCS=>CREATE_INTERNET_ADDRESS
          EXPORTING
            I_ADDRESS_STRING = I_SENDER-SEND_ADDR
            I_ADDRESS_NAME   = I_SENDER-SEND_NAME
          RECEIVING
            RESULT           = SENDER.

        CALL METHOD SEND_REQUEST->SET_SENDER
          EXPORTING
            I_SENDER = SENDER.

      ELSE.

        SAPSENDER = CL_SAPUSER_BCS=>CREATE( SY-UNAME ).
        CALL METHOD SEND_REQUEST->SET_SENDER
          EXPORTING
            I_SENDER = SAPSENDER.

      ENDIF.

*-- ######
      LOOP AT T_MAILRECIPIENTS.

        CLEAR RECIPIENT.

        RECIPIENT = CL_CAM_ADDRESS_BCS=>CREATE_INTERNET_ADDRESS(
                                        T_MAILRECIPIENTS-EMAIL ).

        CALL METHOD SEND_REQUEST->ADD_RECIPIENT
          EXPORTING
            I_RECIPIENT  = RECIPIENT
            I_EXPRESS    = T_MAILRECIPIENTS-EXPRESS
            I_COPY       = T_MAILRECIPIENTS-COPY
            I_BLIND_COPY = T_MAILRECIPIENTS-BLIND_COPY.

      ENDLOOP.

*-- ## ##
      SEND_REQUEST->SET_SEND_IMMEDIATELY( ZGFTP_TRUE ).

      SEND_REQUEST->SET_STATUS_ATTRIBUTES(
                    I_REQUESTED_STATUS = I_REQUESTED_STATUS
                    I_STATUS_MAIL      = I_STATUS_MAIL ).

      CALL METHOD SEND_REQUEST->SEND(
        EXPORTING
          I_WITH_ERROR_SCREEN = ZGFTP_TRUE
        RECEIVING
          RESULT              = SENT_TO_ALL ).

      COMMIT WORK.

    CATCH CX_BCS INTO BCS_EXCEPTION.

      MOVE ZGFTP_E TO E_RETURN-TYPE.

      L_MESSAGE = BCS_EXCEPTION->GET_TEXT( ).

      MESSAGE S125 WITH I_TITLE
                        L_MESSAGE
         INTO E_RETURN-MESSAGE.

      EXIT.

  ENDTRY.

  MOVE ZGFTP_S   TO E_RETURN-TYPE.
  MESSAGE S124 WITH I_TITLE INTO E_RETURN-MESSAGE.

ENDFUNCTION.
