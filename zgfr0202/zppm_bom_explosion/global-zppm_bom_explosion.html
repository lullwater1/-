FORM get_datum_convert USING pv_datum.

  DATA: lv_year    TYPE char04,
        lv_month   TYPE char02,
        lv_day     TYPE char02.

  lv_year  = pv_datum+0(4).
  lv_month = pv_datum+4(2).
  lv_day   = pv_datum+6(2).

  CHECK gs_usr01 IS NOT INITIAL.

  CASE gs_usr01-datfm.
    WHEN gc_1.
      CONCATENATE lv_day lv_month lv_year
        INTO pv_datum.
    WHEN gc_2 OR gc_3.
      CONCATENATE lv_month lv_day lv_year
        INTO pv_datum.
    WHEN OTHERS.
  ENDCASE.
ENDFORM.
