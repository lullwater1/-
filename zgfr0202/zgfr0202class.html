FORM handle_double_click USING p_row
                                p_column.

  READ TABLE gt_outtab INTO gs_outtab INDEX p_row.
  IF sy-subrc = 0.
    CASE p_column.
      WHEN 'MATNR_H'.
        CHECK gs_outtab-matnr_h IS NOT INITIAL.

        SET PARAMETER ID 'MAT' FIELD gs_outtab-matnr_h.
        SET PARAMETER ID 'MXX' FIELD 'K'.
        CALL TRANSACTION 'MM03' AND SKIP FIRST SCREEN.

      WHEN 'IDNRK'.
        CHECK gs_outtab-matnr_h IS NOT INITIAL.

        SET PARAMETER ID 'MAT' FIELD gs_outtab-idnrk.
        SET PARAMETER ID 'MXX' FIELD 'K'.
        CALL TRANSACTION 'MM03' AND SKIP FIRST SCREEN.

      WHEN 'STLNR'.
        CHECK gs_outtab-werks_h IS NOT INITIAL.
        CHECK gs_outtab-matnr_h IS NOT INITIAL.

        SET PARAMETER ID 'MAT' FIELD gs_outtab-matnr_h.
        SET PARAMETER ID 'WRK' FIELD gs_outtab-werks_h.
        SET PARAMETER ID 'CSV' FIELD '1'.
        CALL TRANSACTION 'CS03' AND SKIP FIRST SCREEN.

      WHEN 'ISTLNR'.
        CHECK gs_outtab-endpo IS INITIAL.
        CHECK gs_outtab-werks IS NOT INITIAL.
        CHECK gs_outtab-idnrk IS NOT INITIAL.

        SET PARAMETER ID 'MAT' FIELD gs_outtab-idnrk.
        SET PARAMETER ID 'WRK' FIELD gs_outtab-werks.
        SET PARAMETER ID 'CSV' FIELD '1'.
        CALL TRANSACTION 'CS03' AND SKIP FIRST SCREEN.
    ENDCASE.
  ENDIF.

ENDFORM. " HANDLE_USER_COMMAND

  ENDIF.

*--- ## ##
  LOOP AT gt_outtab INTO gs_outtab.

    " ## ##

                                stlty = gs_outtab-stlty
                                stlnr = gs_outtab-stlnr
                                stlal = gs_outtab-stlal
                       BINARY SEARCH.
    IF sy-subrc = 0.

      gs_outtab-datub_h = gt_head-datub.
      gs_outtab-bmein_h = gt_head-bmein.
      gs_outtab-bmeng_h = gt_head-bmeng.
