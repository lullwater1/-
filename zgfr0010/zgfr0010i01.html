MODULE PERNR_CHECK INPUT.

  FIELD-SYMBOLS <FS_VALUE> TYPE ANY.

  LOOP AT GT_ZKGT0009 WHERE PERNR = GT_KUNAG-PERNR
                        AND DATAB <= GT_KUNAG-IDATE
                        AND DATBI >= GT_KUNAG-IDATE.

    CASE GT_KUNAG-COSIGN.

      WHEN ZGFTP_C.
        ASSIGN GT_ZKGT0009-ORICH TO <FS_VALUE>.

      WHEN ZGFTP_S.
        ASSIGN GT_ZKGT0009-VOLCH TO <FS_VALUE>.

      WHEN ZGFTP_G.
        ASSIGN GT_ZKGT0009-INSCH TO <FS_VALUE>.

      WHEN ZGFTP_I.
        ASSIGN GT_ZKGT0009-INVCH TO <FS_VALUE>.

    ENDCASE.

    MOVE: "GT_ZKGT0009-DATAB TO GT_KUNAG-DATAB,
          GT_ZKGT0009-UNAME TO GT_KUNAG-UNAME,
          GC_ICON           TO GT_KUNAG-BUTTON.
  ENDLOOP.

  IF SY-SUBRC <> 0.
    MESSAGE E006 WITH GT_KUNAG-PERNR.
  ENDIF.

  IF <FS_VALUE> = ZGFTP_FALSE.
    MESSAGE E032.
  ENDIF.

ENDMODULE. " PERNR_CHECK INPUT
