Description: [Global FTA] From Date/To Date Setting
FUNCTION ZGF_SET_DATAB_DATBI.
*"----------------------------------------------------------------------
*"*"Local Interface:
*" IMPORTING
*" VALUE(I_DATE) TYPE DATS DEFAULT SY-DATUM
*" VALUE(I_START_DATE) TYPE NUMC2 OPTIONAL
*" VALUE(I_MONTH) TYPE NUMC2
*" VALUE(I_BUKRS) TYPE BUKRS DEFAULT 'M100'
*" EXPORTING
*" REFERENCE(E_DATAB) TYPE DATAB
*" REFERENCE(E_DATBI) TYPE DATBI
*" EXCEPTIONS
*" MONTH_PARAMETER_NOT_VALID
*"----------------------------------------------------------------------

* Global data declarations

  DATA: L_SDATE TYPE DATS,
        L_EDATE TYPE DATS.

  CLEAR: E_DATAB, E_DATBI.

  IF I_MONTH IS INITIAL.
    RAISE MONTH_PARAMETER_NOT_VALID.
    EXIT.
  ENDIF.

  CASE I_BUKRS.
    WHEN 'M410'.
      L_SDATE = I_DATE+(4) && '0101'.
      L_EDATE = I_DATE+(4) && '1231'.

    WHEN OTHERS.

      IF I_START_DATE IS NOT INITIAL.
        CONCATENATE I_DATE(6) I_START_DATE INTO L_SDATE.
      ELSE.
        MOVE I_DATE TO L_SDATE.
      ENDIF.

      PERFORM RP_CALC_DATE_IN_INTERVAL(ZGF_FORM_SUB_ROUTINE)
              USING   L_SDATE '00' I_MONTH '+' '00'
              CHANGING L_EDATE.

      L_EDATE = L_EDATE - 1.

  ENDCASE.

  MOVE: L_SDATE TO E_DATAB,
        L_EDATE TO E_DATBI.


ENDFUNCTION.
