MODULE MODIFY_MAIN_SCREEN OUTPUT.

  LOOP AT SCREEN.
    CASE G_PRETP.
      WHEN GC_PRE01 OR GC_PRE11.
        IF SCREEN-GROUP1 EQ 'GR2'.
          SCREEN-ACTIVE = 0.
          MODIFY SCREEN.
        ENDIF.
      WHEN GC_PRE12.
        IF SCREEN-GROUP1 EQ 'GR1'.
          SCREEN-ACTIVE = 0.
          MODIFY SCREEN.
        ENDIF.
    ENDCASE.

*-- #### ## TAB ##
    IF G_PRETP EQ GC_PRE01.
      IF SCREEN-NAME EQ 'GS_TAB-TABI'.
        SCREEN-ACTIVE = 0.
        MODIFY SCREEN.
      ENDIF.
    ENDIF.
  ENDLOOP.

ENDMODULE. " MODIFY_MAIN_SCREEN OUTPUT
