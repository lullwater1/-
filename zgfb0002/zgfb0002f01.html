  CLEAR GT_XLS_TAB.

  FIELD-SYMBOLS <FS1> TYPE ANY.

  call function 'ZGF_EXCEL_TO_INTERNAL_TAB'
    EXPORTING
      FILENAME                = P_FNAME
      I_BEGIN_COL             = 1
      I_BEGIN_ROW             = 2
      I_END_COL               = 12
      I_END_ROW               = 500000

    STOP.
  ENDIF.

      CLEAR: <GS_EXCEL>.
    ENDAT.

  ENDLOOP.

FORM DATA_SAVE .

  DATA: LT_ZKGT0002  TYPE TABLE OF ZGFT0002  WITH HEADER LINE,
        LT_ZKGT0002T TYPE TABLE OF ZGFT0002T WITH HEADER LINE.

  DATA: L_OREF TYPE REF TO CX_ROOT,
        L_TEXT TYPE STRING.

  CLEAR G_COUNT.

  FIELD-SYMBOLS <FS1> TYPE ANY.

  DATA LR_SELECTIONS TYPE REF TO CL_SALV_SELECTIONS.

  DATA LT_ROWS  TYPE SALV_T_ROW.
  DATA L_ROW    TYPE I.

  LR_SELECTIONS = GO_ALV->GET_SELECTIONS( ).
  LT_ROWS = LR_SELECTIONS->GET_SELECTED_ROWS( ).

  LOOP AT LT_ROWS INTO L_ROW.

    ADD 1 TO G_COUNT.

    READ TABLE GT_OUTTAB INTO GS_OUTTAB INDEX L_ROW.

    IF SY-SUBRC = 0.

      READ TABLE GT_ZKGT0002 INTO GS_ZKGT0002
                             WITH KEY PFONA = GS_OUTTAB-PFONA
                                      CCNGN = GS_OUTTAB-CCNGN
                                      RSEQ  = GS_OUTTAB-RSEQ
                                      EXHS  = GS_OUTTAB-EXHS.

      IF SY-SUBRC = 0.
*-- #### ## ## ##
        MOVE GS_ZKGT0002 TO LT_ZKGT0002.
        MOVE GS_ZKGT0002-ERDAT TO GS_OUTTAB-ERDAT.
        MOVE GS_ZKGT0002-ERZET TO GS_OUTTAB-ERZET.
        MOVE GS_ZKGT0002-ERNAM TO GS_OUTTAB-ERNAM.

        MOVE-CORRESPONDING GS_OUTTAB TO LT_ZKGT0002.
        ASSIGN LT_ZKGT0002 TO <FS1>.
        PERFORM SET_CREATE_CHANGE_DATA(ZGF_FORM_SUB_ROUTINE)
                                       USING 'M'
                                       CHANGING <FS1>.

      ELSE.
*-- #### ## ## ##
        MOVE-CORRESPONDING GS_OUTTAB TO LT_ZKGT0002.
        ASSIGN LT_ZKGT0002 TO <FS1>.
        PERFORM SET_CREATE_CHANGE_DATA(ZGF_FORM_SUB_ROUTINE)
                                       USING 'A'
                                       CHANGING <FS1>.

      ENDIF.

      APPEND LT_ZKGT0002.
      CLEAR  LT_ZKGT0002.

*-- ###
      MOVE-CORRESPONDING GS_OUTTAB TO LT_ZKGT0002T.
      MOVE SY-LANGU                TO LT_ZKGT0002T-SPRAS.
      APPEND LT_ZKGT0002T.
      CLEAR  LT_ZKGT0002T.

    ENDIF.

  ENDLOOP.

  IF SY-SUBRC <> 0.
    MESSAGE I005 DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.

  TRY.
      CLEAR L_TEXT.

      MODIFY ZGFT0002  FROM TABLE LT_ZKGT0002.
      MODIFY ZGFT0002T FROM TABLE LT_ZKGT0002T.

      COMMIT WORK.
      MESSAGE I003 WITH G_COUNT.
      MOVE ICON_LED_GREEN TO GS_OUTTAB-ICON.
      MODIFY GT_OUTTAB FROM GS_OUTTAB TRANSPORTING ICON
                                      WHERE ICON <> ICON_LED_GREEN.

    CATCH CX_SY_OPEN_SQL_DB INTO L_OREF.
      L_TEXT = L_OREF->GET_TEXT( ).

      ROLLBACK WORK.
      MESSAGE I004 WITH L_TEXT DISPLAY LIKE 'E'.
      MOVE ICON_LED_RED TO GS_OUTTAB-ICON.
      MODIFY GT_OUTTAB FROM GS_OUTTAB TRANSPORTING ICON
                                      WHERE ICON <> ICON_LED_RED.
  ENDTRY.

  CALL METHOD GO_ALV->REFRESH
    EXPORTING
      REFRESH_MODE = IF_SALV_C_REFRESH=>SOFT.

ENDFORM. " DATA_SAVE
