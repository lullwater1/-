FORM GET_DATA .

  SELECT A~*,
         B~REMARK,
         C~AGR_NAME,
         D~BUTXT
    FROM ZTMDM6110 AS A INNER JOIN AGR_USERS AS C
                                  ON A~EMP_ID = C~UNAME
                        LEFT OUTER JOIN ZGFT2005 AS B
                                  ON C~UNAME = B~EMP_ID AND
                                     C~AGR_NAME = B~AGR_NAME
                        LEFT OUTER JOIN T001 AS D
                                  ON A~ERPC_CD = D~BUKRS
    INTO CORRESPONDING FIELDS OF TABLE @GT_OUTTAB
    WHERE A~ERPC_CD IN @S_BUKRS
      AND A~EMP_ID IN @S_EID
      AND C~AGR_NAME IN @S_AUTH.

  SORT GT_OUTTAB BY AGR_NAME.

*  SORT GT_OUTTAB BY AGR_NAME

*                    EMP_ID.
*  DELETE ADJACENT DUPLICATES FROM GT_OUTTAB COMPARING AGR_NAME
*                                                      ERPC_CD
*                                                      EMP_ID.
  GV_SAVED = ABAP_TRUE.   " Initial state

  IF GT_OUTTAB[] IS INITIAL.
    MESSAGE S000 WITH TEXT-M01 DISPLAY LIKE 'E'.
  ENDIF.

FORM SAVE_DATA .

  DATA: LT_OUTTAB   TYPE TABLE OF GTY_OUTTAB,
        LS_ZGFT2005 TYPE ZGFT2005,
        LT_ZGFT2005 TYPE TABLE OF ZGFT2005.

  MOVE-CORRESPONDING GT_OUTTAB[] TO LT_OUTTAB[].
  DELETE LT_OUTTAB[] WHERE CHANGED IS INITIAL.

  MOVE-CORRESPONDING LT_OUTTAB[] TO LT_ZGFT2005[].

  IF LT_ZGFT2005[] IS INITIAL.
    MESSAGE S000 WITH TEXT-M07 DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.

  GET TIME.
  LS_ZGFT2005-ERDAT = LS_ZGFT2005-AEDAT = SY-DATUM.
  LS_ZGFT2005-ERZET = LS_ZGFT2005-AEZET = SY-UZEIT.
  LS_ZGFT2005-ERNAM = LS_ZGFT2005-AENAM = SY-UNAME.
  MODIFY LT_ZGFT2005 FROM LS_ZGFT2005 TRANSPORTING ERDAT AEDAT
                                                   ERZET AEZET
                                                   ERNAM AENAM
                                      WHERE EMP_ID IS NOT INITIAL.


  MODIFY ZGFT2005 FROM TABLE LT_ZGFT2005.
  MESSAGE S000 WITH TEXT-M06.

ENDFORM.
