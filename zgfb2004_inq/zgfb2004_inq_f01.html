FORM get_data .

  DATA: lt_outtab_top TYPE TABLE OF ty_outtab_top.

  SELECT a~bukrs,
         a~ccno,
         a~cctype as cotype,
         a~werks,
         a~lifnr,
         b~name1  AS lifnm,
         c~f_name,
         a~cust_ccno
    INTO CORRESPONDING FIELDS OF TABLE @lt_outtab_top
    FROM zgft8000 AS a
    LEFT OUTER JOIN lfa1 AS b
      ON a~lifnr = b~lifnr
    LEFT OUTER JOIN zgft8002 AS c
      ON a~bukrs = c~bukrs
     AND a~ccno  = c~ccno
    WHERE a~bukrs    = @p_bukrs
      AND a~ccno     IN @s_ccno
      AND a~werks    IN @s_werks
      AND a~lifnr    IN @s_lifnr
      AND a~idate    IN @s_idate
      AND EXISTS (
            SELECT 1
              FROM zgft8001 AS x
             WHERE x~bukrs = a~bukrs
               AND x~ccno  = a~ccno
               AND x~mtroh IN @s_matnr
               AND x~zdeno IN @s_zdeno
               AND x~budat IN @s_budat ).

  gt_outtab_top[] = lt_outtab_top[].

ENDFORM.
