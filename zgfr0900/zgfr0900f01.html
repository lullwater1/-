*
*  IF SY-SUBRC = 0.
*    MESSAGE I016 WITH TEXT-T16 DISPLAY LIKE 'E'.

*
**-- ### ##
*  LT_OUTTAB[] = GT_OUTTAB[].
*
*  SORT LT_OUTTAB BY BUKRS LIFNR.
*  DELETE ADJACENT DUPLICATES FROM LT_OUTTAB COMPARING BUKRS LIFNR.

*  IF SY-SUBRC = 0.
*    MESSAGE I048 WITH TEXT-T16 DISPLAY LIKE 'E'.

  LOOP AT LT_ROWS INTO LS_ROWS.

      MOVE-CORRESPONDING GT_OUTTAB TO LS_ZKGT7001.

*      ADD 1 TO LS_ZKGT7001-SEQ.

      ASSIGN LS_ZKGT7001 TO <FS1>.
      PERFORM SET_CREATE_CHANGE_DATA(ZGF_FORM_SUB_ROUTINE)
                                     USING 'A'
                                           <FS1>.
      TRY.
          CLEAR L_TEXT.
          MODIFY ZGFT7001 FROM LS_ZKGT7001.
        CATCH CX_SY_OPEN_SQL_DB INTO L_OREF.
          L_TEXT = L_OREF->GET_TEXT( ).
      ENDTRY.

        EXIT.
      ENDIF.

    ENDIF.

    LOOP AT GT_OUTTAB.
*      ADD 1 TO GT_OUTTAB-SEQ.
      REFRESH GT_OUTTAB-CELLTAB[].
      MODIFY GT_OUTTAB.
    ENDLOOP.

    SELECT * FROM ZGFT7001
      INTO TABLE GT_ZKGT7001
       FOR ALL ENTRIES IN GT_OUTTAB
     WHERE BUKRS = GT_OUTTAB-BUKRS
       AND LIFNR = GT_OUTTAB-LIFNR.

  ELSE.
    ROLLBACK WORK.
    MESSAGE I004 DISPLAY LIKE ZGFTP_E.
  ENDIF.

ENDFORM. " DATA_SAVE

FORM CONTACT_SELECT .

  DATA : LT_ROWS TYPE LVC_T_ROW,

        LT_ZKGT7000 TYPE TABLE OF ZGFT7000 WITH HEADER LINE.

      ET_INDEX_ROWS = LT_ROWS[].

  IF LT_ROWS[] IS INITIAL.
    MESSAGE S005 DISPLAY LIKE ZGFTP_E.
    EXIT.
  ENDIF.

  DESCRIBE TABLE LT_ROWS LINES L_LINES.

  IF L_LINES > 1.
    MESSAGE S044 DISPLAY LIKE ZGFTP_E.
    EXIT.
  ENDIF.

                          LIFNR ASCENDING
                          PERNR ASCENDING
                          F_SEQ DESCENDING.
      DELETE ADJACENT DUPLICATES FROM LT_ZKGT7000 COMPARING BUKRS
                                                            LIFNR
                                                            PERNR.
      LOOP AT LT_ZKGT7000.
        IF LT_ZKGT7000-DELE = ''.
          MOVE-CORRESPONDING LT_ZKGT7000 TO GT_CONTACT.
          APPEND GT_CONTACT.
        ENDIF.
      ENDLOOP.

    ENDIF.

  ENDLOOP.

  IF GT_CONTACT[] IS INITIAL.
    MESSAGE S006 DISPLAY LIKE ZGFTP_E.
    EXIT.
  ENDIF.

  CALL SCREEN 160 STARTING AT 10 3
                  ENDING   AT 170 20.
