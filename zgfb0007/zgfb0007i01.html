MODULE user_command_0100 INPUT.
  DATA l_dummy    TYPE c LENGTH 100.
  DATA l_check(1) TYPE c.
  DATA lt_sel     type lvc_t_row.

  CLEAR g_code.
  g_code = ok_code.

  CLEAR gs_outtab.

  CASE g_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'SAVE'.
      CALL METHOD g_grid1->check_changed_data.

      IF gt_outtab[] IS NOT INITIAL.
        MESSAGE s002 INTO l_dummy.
        PERFORM confirm_step(zgf_form_sub_routine) USING    l_dummy
                                                   CHANGING g_answer.

        CHECK g_answer = '1'.

        LOOP AT gt_outtab INTO gs_outtab.

          PERFORM empty_check  CHANGING l_check.
          IF l_check = '0'.
            REJECT.
          ENDIF.
          PERFORM pfona_check  CHANGING l_check.
          IF l_check = '0'.
            REJECT.
          ENDIF.
          MODIFY gt_outtab FROM gs_outtab.
          CLEAR gs_outtab.
        ENDLOOP.
        CALL METHOD g_grid1->get_selected_rows
          IMPORTING
            et_index_rows = lt_sel.
        PERFORM refresh_display.
        PERFORM data_save using lt_sel.
      ELSE.
*        MESSAGE I000 WITH '### #### ####.'
        MESSAGE i000 WITH 'No data.'
                     DISPLAY LIKE 'E'.
      ENDIF.

  ENDCASE.

ENDMODULE. " USER_COMMAND_0100 INPUT
